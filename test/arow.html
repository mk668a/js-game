script

      <script type="text/javascript">
        var name = "taro",
          age = 18
        console.log("リテラル", name, age);

        let gettriangle = function (base, height) {
          return base * height / 2
        }
        console.log("三角形の面積は" + gettriangle(10, 2));

        console.log("アロー関数");
        let gettriangle2 = (base, height) => {
          return base * height / 2
        }
        console.log("三角形の面積は" + gettriangle2(10, 2));

        console.log("より簡単に");
        let gettriangle3 = (base, height) => base * height / 2
        console.log("三角形の面積は" + gettriangle3(10, 2));

        console.log("引数が一つの場合");
        let getcircle = radius => radius * radius * math.pi
        console.log(getcircle(10));

        console.log("引数がない場合");
        let show = () => "hello,world"
        console.log(show());
      </script>
